add_namespace = rgs

# Iran capitulation - event to make peace
country_event = {
	id = rgs.1
	title = rgs.1.t
	desc = rgs.1.d
	picture = GFX_news_event_iran

	fire_only_once = yes
	
	trigger = {
		tag = PER
		surrender_progress > 0.5
		OR = {
			has_war_with = ENG
			has_war_with = SOV
		}
	}

	mean_time_to_happen = {
		days = 2
	}
	
	option = { # Accept demands
		name = rgs.1.a
		SOV = {
			transfer_state = 419
			transfer_state = 420
			transfer_state = 266
		}
		ENG = {
			transfer_state = 412
			transfer_state = 413
		}
		every_country = {
			limit = {
				is_in_faction_with = SOV
				has_war_with = PER
			}
			white_peace = PER
		}
		every_country = {
			limit = {
				is_in_faction_with = ENG
				has_war_with = PER
			}
			white_peace = PER
		}
		leave_faction = yes
		news_event = { id = rgs.2 }
	}
}

# End of Iraqi concessions - British event
country_event = {
	id = rgs.3
	title = rgs.3.t
	desc = rgs.3.d
	picture = GFX_report_event_journalists_speech

	fire_only_once = yes
	
	trigger = {
		tag = ENG
		
		# Athens (47) and the Crete (182) are lost
		GRE = {
			NOT = { controls_state = 47 }
			NOT = { controls_state = 182 }
		}
		
		# Iraq is not already at war with England or one of its subject
		IRQ = {
			NOT = {
				has_war_with = ROOT
			}
			NOT = {
				is_subject_of = ROOT
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}
	
	option = { # Iraq stop oil concessions
		name = rgs.3.a
		remove_resource_rights = 291
		remove_resource_rights = 676
		add_opinion_modifier = {
			target = IRQ
			modifier = sanctions
		}
		create_wargoal = {
			type = puppet_wargoal_focus
			target = IRQ
		}
		news_event = { id = rgs.4 }
	}
}
